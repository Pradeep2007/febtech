// Copy these rules to your Firestore Database Rules tab in Firebase Console
// Go to: Firebase Console → Firestore Database → Rules → Paste this content → Publish

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone to create contact messages
    match /contacts/{contactId} {
      allow create: if true;
      allow read: if false; // Only allow creation, not reading
    }
    
    // Allow full access to products collection (for admin functionality)
    match /products/{productId} {
      allow read, write: if true; // Allow all operations for now
    }
    
    // Allow read access to other collections (testimonials, partners, etc.)
    match /testimonials/{testimonialId} {
      allow read: if true;
      allow write: if true; // Allow adding testimonials
    }
    
    match /partners/{partnerId} {
      allow read: if true;
      allow write: if true;
    }
    
    match /team/{teamId} {
      allow read: if true;
      allow write: if true;
    }
    
    // Test collection for debugging
    match /test/{testId} {
      allow read, write: if true;
    }
  }
}
